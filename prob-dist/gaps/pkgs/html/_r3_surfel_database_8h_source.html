262 {
263  // Check whether block needs to be written
264  if (block->file_read_count == 1) {
265  if (!InternalReleaseBlock(block)) return 0;
266  }
267 
268  // Decrement reference count
269  block->file_read_count--;
270 
271  // Check if delete pending
272  if (block->file_read_count == 0) {
273  if (block->flags[R3_SURFEL_BLOCK_DELETE_PENDING_FLAG]) {
274  RemoveBlock(block);

277  }
278 
279  // Return success
280  return 1;
