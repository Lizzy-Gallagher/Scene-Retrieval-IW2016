357 inline R3Triad R3Grid::
358 WorldPrincipleAxes(const R3Point *world_centroid, RNScalar *variances) const
359 {
360  // Return principle axes in world coordinates
361  R3Point grid_centroid = (world_centroid) ? GridPosition(*world_centroid) : GridCentroid();
362  R3Triad axes = GridPrincipleAxes(&grid_centroid, variances);
363  axes.Transform(grid_to_world_transform);
364  variances[0] *= grid_to_world_scale_factor;
365  variances[1] *= grid_to_world_scale_factor;
366  variances[2] *= grid_to_world_scale_factor;
367  return axes;
368 }
