367 inline R3Affine R3PlanarGrid::
368 WorldToGridTransformation(void) const
369 {
370  // Create 3D transformation for plane->grid
371  R3Matrix m2 = grid.WorldToGridTransformation().Matrix();
372  R4Matrix m3(m2[0][0], m2[0][1], 0, m2[0][2],
373  m2[1][0], m2[1][1], 0, m2[1][2],
374  0, 0, 1, 0,
375  0, 0, 0, 1);
376 
377  // Return transformation from 3D world to 2D grid
378  R3Affine result = R3identity_affine;
379  result.Transform(R3Affine(m3));
380  result.Transform(transformation);
381  return result;
382 }

386 inline R3Affine R3PlanarGrid::
387 GridToWorldTransformation(void) const
388 {
389  // Create 3D transformation for grid->plane
390  R3Matrix m2 = grid.GridToWorldTransformation().Matrix();
391  R4Matrix m3(m2[0][0], m2[0][1], 0, m2[0][2],
392  m2[1][0], m2[1][1], 0, m2[1][2],
393  0, 0, 1, 0,
394  0, 0, 0, 1);
395 
396  // Return transformation from 2D grid to 3D world
397  R3Affine result = R3identity_affine;
398  result.Transform(transformation.Inverse());
399  result.Transform(R3Affine(m3));
400  return result;
401 }

415 inline RNScalar R3PlanarGrid::
416 GridToWorldScaleFactor(void) const
417 {
418  // Return scale from grid->world
419  RNScalar s = transformation.ScaleFactor();

422  return grid_to_plane_scale * plane_to_world_scale;
